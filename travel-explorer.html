        function renderDurationChart(data) {

            const ctx = document.getElementById('durationChart').getContext('2d');

            if (charts.duration) charts.duration.destroy();

            

            const durationBuckets = {

                '5-7 Days': { min: 5, max: 7, count: 0 },

                '8-10 Days': { min: 8, max: 10, count: 0 },

                '11-13 Days': { min: 11, max: 13, count: 0 },

                '14-16 Days': { min: 14, max: 16, count: 0 },

                '17+ Days': { min: 17, max: Infinity, count: 0 }

            };


            data.forEach(trip => {

                const duration = (new Date(trip.endDate) - new Date(trip.startDate)) / (1000 * 60 * 60 * 24) + 1;

                for (const bucket in durationBuckets) {

                    if (duration >= durationBuckets[bucket].min && duration <= durationBuckets[bucket].max) {

                        durationBuckets[bucket].count++;

                        break;

                    }

                }

            });


            const labels = Object.keys(durationBuckets);

            const chartData = labels.map(label => durationBuckets[label].count);


            charts.duration = new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: labels,

                    datasets: [{

                        label: 'Number of Trips',

                        data: chartData,

                        backgroundColor: '#4A4A4A'

                    }]

                },

                options: {

                    indexAxis: 'y',

                    responsive: true,

                    maintainAspectRatio: false,

                    scales: { x: { beginAtZero: true } },

                    plugins: { legend: { display: false } },

                        if (elements.length > 0) {

                            const label = labels[elements[0].index];

        async function callGeminiAPI(payload, maxRetries = 5) {

            for (let attempt = 0; attempt < maxRetries; attempt++) {

                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    if (response.status === 429 && attempt < maxRetries - 1) {

                        const delay = Math.pow(2, attempt) * 1000;

                        await new Promise(resolve => setTimeout(resolve, delay));

                        continue;

                    }


                    if (!response.ok) {

                        throw new Error(`HTTP error! status: ${response.status}`);

                    }


                    const result = await response.json();

                    const candidate = result.candidates?.[0];


                    if (candidate && candidate.content?.parts?.[0]?.text) {

                        const text = candidate.content.parts[0].text;

                        let sources = [];

                        const groundingMetadata = candidate.groundingMetadata;

                        if (groundingMetadata && groundingMetadata.groundingAttributions) {

                            sources = groundingMetadata.groundingAttributions

                                .map(attribution => ({

                                    uri: attribution.web?.uri,

                                    title: attribution.web?.title,

                                }))

                                .filter(source => source.uri && source.title);

                        }

                        return { text, sources };

                    } else {

                        return { text: "Could not generate content. Please try again.", sources: [] };

                    }


                } catch (error) {

                    if (attempt === maxRetries - 1) {

                        return { text: `API call failed: ${error.message}`, sources: [] };

                    }

                    const delay = Math.pow(2, attempt) * 1000;

                    await new Promise(resolve => setTimeout(resolve, delay));

                }

            }

        }

            const button = outputDiv.previousElementSibling;


            // Toggle visibility if already showing

            if (!outputDiv.classList.contains('hidden') && contentDiv.textContent.length > 0) {

                outputDiv.classList.add('hidden');

                button.textContent = '✨ Ask Gemini: Destination Quick Facts';

            }


            outputDiv.classList.remove('hidden');

            button.textContent = 'Hide Gemini Facts';


            contentDiv.innerHTML = '<p class="text-gray-500 flex items-center">Loading quick facts... <svg class="animate-spin ml-2 h-4 w-4 text-indigo-500" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg></p>';


            const systemPrompt = "You are a highly knowledgeable travel expert. Your task is to provide three concise, interesting, and up-to-date facts about the primary destination of the requested trip. Use bullet points. Ensure the information is grounded in search results.";

            const userQuery = `Provide three quick facts about the primary destinations for the alumni trip: "${tripName}".`;


            const payload = {

                contents: [{ parts: [{ text: userQuery }] }],

                tools: [{ "google_search": {} }],

                systemInstruction: {

                    parts: [{ text: systemPrompt }]

                },

            const result = await callGeminiAPI(payload);

        function displayGeminiResult(outputDiv, text, sources) {

            const contentDiv = outputDiv.querySelector('.gemini-content');

            const formattedText = text

                .replace(/\* /g, '<span class="text-indigo-600 font-bold mr-1">•</span>')

                .replace(/(\r\n|\n|\r)/gm, '<br>');

            

            contentDiv.innerHTML = formattedText;


            if (sources && sources.length > 0) {

                let sourcesHtml = '<div class="mt-3 pt-3 border-t border-gray-200 text-xs text-gray-500">';

                sourcesHtml += 'Source Citations:';

                sources.forEach((source, index) => {

                    if (source.uri && source.title) {

                        sourcesHtml += `<br> ${index + 1}. <a href="${source.uri}" target="_blank" class="text-indigo-600 hover:underline">${source.title.substring(0, 50)}...</a>`;

                    }

                });

                sourcesHtml += '</div>';

                contentDiv.innerHTML += sourcesHtml;

            }

        }
